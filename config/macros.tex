\NewDocumentCommand{\TinyGraph}{O{plain} O{1.0}}{%
  \begin{tikzpicture}[scale=#2,every node/.style={transform shape}]
    \coordinate (a) at (-1.2,0.9);
    \coordinate (b) at (-0.2,1.4);
    \coordinate (c) at (0.9,1.0);
    \coordinate (d) at (1.6,0.0);
    \coordinate (e) at (0.6,-0.6);
    \coordinate (f) at (-0.6,-0.5);
    \coordinate (adv) at (-1.2,-0.1);
    \draw[gedge] (a)--(b)--(c)--(d)--(e)--(f)--(adv)--(a);
    \draw[gedge] (b)--(e); \draw[gedge] (c)--(f);
    \IfEqCase{#1}{%
      {plain}{}%
      {addedge}{\draw[addedge] (adv)--(d);}%
      {cutedge}{\draw[cutedge] (adv)--(a);}%
      {fake}{\node[gfake] (fk) at (-0.1,0.1) {};\draw[addedge] (adv)--(fk);}%
    }[\relax]
    \node[gnode] at (a) {};
    \node[gnode] at (b) {};
    \node[gnode] at (c) {};
    \node[gnode] at (d) {};
    \node[gnode] at (e) {};
    \node[gnode] at (f) {};
    \node[gadv] at (adv) {};
  \end{tikzpicture}%
}
