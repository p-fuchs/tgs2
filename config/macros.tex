\NewDocumentCommand{\ToyGraph}{O{clean} O{1.0}}{%
  \begin{tikzpicture}[scale=#2,every node/.style={transform shape}]
    \coordinate (a) at (0,1.5);
    \coordinate (b) at (1.2,2.1);
    \coordinate (c) at (1.0,0.6);
    \coordinate (d) at (0.15,-0.8);
    \coordinate (e) at (2.3,1.2);
    \coordinate (t) at (2.45,-0.45);
    \draw[graphedge] (a)--(b)--(c)--(a)--(d)--(c)--(e)--(b);
    \draw[graphedge] (c)--(t);
    \IfEqCase{#1}{%
      {perturbed}{\draw[attackedge] (d)--(t) node[midway,below,font=\tiny\bfseries] {+};\draw[attackedge] (e)--(t) node[midway,right,font=\tiny\bfseries] {+};}%
      {rewired}{\draw[fadededge] (c)--(t);\draw[attackedge] (d)--(t) node[midway,below,font=\tiny\bfseries] {+};\draw[PresThreeAttack,line width=0.9pt] ($(c)!0.44!(t)+(-0.10,0.10)$)--($(c)!0.56!(t)+(0.10,-0.10)$);\draw[PresThreeAttack,line width=0.9pt] ($(c)!0.44!(t)+(0.10,-0.10)$)--($(c)!0.56!(t)+(-0.10,0.10)$);}%
      {defended}{\draw[fadededge] (d)--(t);\draw[defenseedge] (e)--(t);\node[draw=PresThreeDefense,fill=PresThreeDefenseSoft,rounded corners=2pt,font=\tiny,inner sep=1.5pt] at (1.65,-0.95) {shield};}%
    }[\relax]
    \node[graphnode,fill=PresThreeBlue!18] at (a) {};
    \node[graphnode,fill=PresThreeTeal!24] at (b) {};
    \node[graphnode,fill=PresThreeRose!24] at (c) {};
    \node[graphnode inf] at (d) {I};
    \node[graphnode inf] at (e) {I};
    \node[graphnode target] at (t) {T};
    \FeatureBadge{(-0.35,1.88)}{PresThreeMuted}
    \FeatureBadge{(1.05,2.55)}{PresThreeMuted}
    \FeatureBadge{(2.88,1.60)}{PresThreeMuted}
    \IfEqCase{#1}{%
      {perturbed}{\FeatureBadge{(2.88,-0.05)}{PresThreeAttack}}%
      {rewired}{\FeatureBadge{(2.88,-0.05)}{PresThreeAttack}}%
      {defended}{\FeatureBadge{(2.88,-0.05)}{PresThreeDefense}}%
    }[\FeatureBadge{(2.88,-0.05)}{PresThreeMuted}]
  \end{tikzpicture}%
}
